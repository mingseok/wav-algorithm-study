```python
def solution(brown, yellow):
    answer = 0
    b = brown
    y = yellow
    by = int(b) + int(y)
    imsi = []
    nanu = 2
    fin = []
    while nanu < by//2:
        
        if by % nanu == 0 :
            imsi.append([ nanu, by // nanu])
        nanu = nanu + 1
    for ix, iy in imsi:
        if (ix-2) * (iy - 2) == yellow:
            if ix > iy :
                fin = [ix,iy]
            else :
                fin = [iy,ix]
            break
    return fin
```
O(n)
