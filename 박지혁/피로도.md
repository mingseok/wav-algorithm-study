```python
import itertools

def solution(k, dungeons):
    answer = 0

    def dfs(now_k, count, visited):
        nonlocal answer

        if count > answer:
            answer = count

        for i in range(len(dungeons)):
            if visited[i] == False:
                if now_k >= dungeons[i][0]:
                    visited[i] = True
                    dfs(now_k - dungeons[i][1], count + 1, visited)
                    visited[i] = False

    visited = [False] * len(dungeons)
    dfs(k, 0, visited)

    return answer
```
O(n!)
