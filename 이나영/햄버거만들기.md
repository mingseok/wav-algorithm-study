## 햄버거만들기(Java)
## https://school.programmers.co.kr/learn/courses/30/lessons/133502

```JAVA
import java.util.Stack;

class Solution {
    public int solution(int[] ingredient) {
        int answer = 0;
        Stack<Integer> berger = new Stack<>();
        
        for (int i = 0; i < ingredient.length; i++) {
            berger.push(ingredient[i]);
            
            if (berger.size() >= 4) {
                int a = berger.get(berger.size() - 4);
                int b = berger.get(berger.size() - 3);
                int c = berger.get(berger.size() - 2);
                int d = berger.get(berger.size() - 1);
                
                if (a == 1 && b == 2 && c == 3 && d == 1) {
                    berger.pop();
                    berger.pop();
                    berger.pop();
                    berger.pop();
                    answer++;
                }
            }
        }
        return answer;
    }
}
```
