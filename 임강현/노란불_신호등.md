class Solution {
    int MAX_TIME = 1;
    int[][] signals;
    public int solution(int[][] _signals) {
        signals = _signals;
        int n = signals.length;
        int answer = 0;
        int[] cycle = new int[n];
        MAX_TIME = lcm(n, cycle);
        for (int t = 1; t <= MAX_TIME; t++) {
            boolean allYellow = true;
            for (int i = 0; i < n; i++) {
                int mod = t % cycle[i];
                int green = signals[i][0];
                int yellow = signals[i][1];
                if (!(mod > green && mod <= green + yellow)) {
                    allYellow = false;
                    break;
                }
            }
            if (allYellow) return t;
        }
        return -1;
    }
    private int lcm(int n, int[] cycle) {
        int result = 1;
        for (int i = 0; i < n; i++) {
            cycle[i] += signals[i][0] + signals[i][1] + signals[i][2];
            result = lcm(result, cycle[i]);
        }
        return result;
    }
    private int lcm(int a, int b) {
        return a / gcd(a, b) * b;
    }
    private int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
}
