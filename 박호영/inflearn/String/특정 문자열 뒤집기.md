package inflearn.string;

import java.util.Scanner;
import java.util.Deque;
import java.util.ArrayDeque;
import java.util.Map;

public class Main5 {

    public String solution(String str){
        Deque<Map.Entry<Integer, Character>> queue = new ArrayDeque<>();
        
        char[] chs = str.toCharArray();
        for(int i = 0 ; i < chs.length ; i++){
            if (Character.isAlphabetic(chs[i])) {
                queue.offerLast(Map.entry(i,chs[i]));
            }
        }
        while (!queue.isEmpty()) {
            Map.Entry<Integer, Character> front = queue.pollFirst();
            Map.Entry<Integer,Character> back = queue.pollLast();
            if(front != null && back != null){
                chs[front.getKey()] = back.getValue();
                chs[back.getKey()] = front.getValue();
            }else{
                queue.clear();
            }
        }
        return new String(chs);
    }
    public static void main(String[] args){
        Main5 m = new Main5();
        Scanner in = new Scanner(System.in);
        String str = in.nextLine();
        System.out.println(m.solution(str));
    }
}
