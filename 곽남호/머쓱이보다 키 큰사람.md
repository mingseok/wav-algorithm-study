### 머쓱이보다 키 큰사람

```java

class Solution {
    public int solution(int[] array, int height) {
        int answer = 0;
        
        for (int i = 0 ; i < array.length - 1; i++){
            for (int j = 0; j < array.length -1 -i; j++){
                if(array[j] > array [j+1]){
                    int temp = array[j];
                    array[j] = array[j+1];
                    array[j+1] = temp;
                }
            }
        }
        
        for (int i=0; i< array.length; i++){
            if(array[i] > height){
                answer++;
            }
        }
        
        return answer;
    }
}
```
