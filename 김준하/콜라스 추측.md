```java
/**
 * 626331을 입력받았을 때, 오버플로우가 발생되어 long 으로 형변환하여 작업
 */
class Solution {
    public int solution(int num) {
        long convertNum = num;
        int count = 0;

        while (convertNum != 1) {
            if (convertNum % 2 == 0) convertNum /= 2;
            else convertNum = convertNum * 3 + 1;
            count++;
            if (count >= 500) {
                count = -1;
                break;
            }
        }
        return count;
    }
}
```
