## 야근 지수

- https://school.programmers.co.kr/learn/courses/30/lessons/12927

```java
import java.util.Arrays;
import java.util.Collections;
import java.util.PriorityQueue;

/**
 * 시간(n)이 지날 때마다 works를 재정렬하게 만들었더니 효율성 테스트에서 실패가 떴다
 * -> 우선순위 큐(Priority Queue) 사용으로 하나의 배열에서 우선순위가 높은(Collections.reverseOrder(), 값이 높은) 숫자부터 작업이 이루어질 수 있도록 처리하엿다.
 * PriorityQueue는 꺼낼 때(peek, poll)만 우선순위를 따진다
 */

class Solution {
    public long solution(int n, int[] works) {
        PriorityQueue<Integer> worksQueue = new PriorityQueue<>(Collections.reverseOrder()); // 큰 값이 우선순위를 가지도록 
        for (int x : works) worksQueue.offer(x);
        long result = 0;

        while (n != 0) {
            if(worksQueue.peek()==0) return 0;
            worksQueue = work2(worksQueue);
            n--;
        }
        for (long x : worksQueue) result += x*x;
        return result;
    }

    private int[] work(int[] works) { // 너무 시간이 오래걸림
        Arrays.sort(works);
        if(works[works.length -1] == 0) return works;
        works[works.length -1] -= 1;
        return works;
    }

    private PriorityQueue<Integer> work2(PriorityQueue<Integer> worksQueue) {
        worksQueue.offer(worksQueue.poll() - 1);
        // worksQueue.remove(); // 첫 번째 값 제거
        return worksQueue;
    }
}
```
